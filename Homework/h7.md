import string

cipher_hex = "1b37373331363f78151b7f2b783431333d78397828372d363c78373e783a393b3736"
cipher = bytes.fromhex(cipher_hex)

def score(text):
    freq = " etaoinshrdlu"
    return sum(text.count(c) for c in freq)

best_score = 0
best_text = ""
best_key = None

for key in range(256):
    decoded = bytes(c ^ key for c in cipher)
    try:
        decoded_text = decoded.decode("ascii")
    except:
        continue

    s = score(decoded_text.lower())

    if s > best_score:
        best_score = s
        best_text = decoded_text
        best_key = key

print("key:", best_key)
print("message:", best_text)

cipher_lines = open("data.txt").read().splitlines()

    freq = " etaoinshrdlu"
    return sum(text.count(c) for c in freq)

best_score = 0
best_message = ""

        s = score(text.lower())

            best_message = text
            best_key = key

print("line:", best_line)
